   public function ios_push($deviceToken,$aps,$custom_data)
     {
       // Put your private key's passphrase here:å¯†è¯­
       $passphrase = '123456';
       $keys = __DIR__."/../storage/app/hopewell_push_prod.pem";
       $ctx = stream_context_create();
       stream_context_set_option($ctx, 'ssl', 'local_cert', $keys);
       stream_context_set_option($ctx, 'ssl', 'passphrase', $passphrase);

       $gateway = 'ssl://gateway.push.apple.com:2195';
       // $gateway = 'ssl://gateway.sandbox.push.apple.com:2195';
       // Open a connection to the APNS server
       $fp = stream_socket_client(
           $gateway, $err,
           $errstr, 60, STREAM_CLIENT_CONNECT|STREAM_CLIENT_PERSISTENT, $ctx);

       if (!$fp)
         exit("Failed to connect: $err $errstr" . PHP_EOL);

       // Encode the payload as JSON
       $date = array(
         'aps'=>$aps,
         'app_link_code'=>$custom_data['app_link_code'],
         'channel_id'=> isset($custom_data['channel_id'])  ? $custom_data['channel_id'] : "",
         'message_id'=> isset($custom_data['message_id']) ? $custom_data['message_id'] : "",
         'news_id'=> isset($custom_data['newsId']) ? $custom_data['newsId'] : "",
         'title'=> isset($custom_data['title']) ? $custom_data['title'] : "",
       );
       $payload = json_encode($date);

       // Build the binary notification
       $msg = chr(0) . pack('n', 32) . pack('H*', $deviceToken) . pack('n', strlen($payload)) . $payload;
       // Send it to the server
       $result = fwrite($fp, $msg, strlen($msg));
       echo "IOSPUSH:deviceToken:".$deviceToken."   aps:".$payload."  result:".json_encode($result);
       // $log->info("Key:".$ios_push_file." IOSPUSH:deviceToken:".$deviceToken."   aps:".$payload."  result:".json_encode($result));

       // Close the connection to the server
         // @socket_close($fp);
       fclose($fp);
       if (!$result)
         return false;
       else
         return true;
     }
